#include <stdio.h>
#define true 1
#define false 0

typedef int bool;

bool get_str_len(char* str) {
  for (int i = 0; true; i++) {
    if (str[i] == '\0') {
      return i;
    }
  }
}

bool str_equals(char* str1, char* str2) {
  if (str1 == str2) {
    return true;
  }
  int len_of_str1 = get_str_len(str1);
  int len_of_str2 = get_str_len(str2);

  if (len_of_str1 != len_of_str2) {
    return false;
  }
  for (int i = 0; i < len_of_str1; i++) {
    if (str1[i] != str2[i]) {
      return false;
    }
  }
  return true;
}

bool str_part_equals(char* str1, int start, int end, char* str2) {
  int j = 0;
  for (int i = start; i < end; i++) {
    if (str1[i] != str2[j]) {
      return false;
    }
    j++;
  }
  return true;
}

		}
	}
}
bool get_index_of_str(char* str1, char* str2) {

	int len_of_str1 = get_str_len(str1);//3
	int len_of_str2 = get_str_len(str2);//1
	
	if (len_of_str1 % 2 != 0){
	for (int i = len_of_str1%2; i <= len_of_str1 % 2+1; i++) {
		for (int j = 0; j <= len_of_str2; j++){
			if (str1[i] == str2[i]) {
				return true;
			}
		}
			
		}
	
	}
	else if(len_of_str1 % 2 == 0){
		for (int i = len_of_str1 % 2-1; i <= len_of_str1 % 2 ; i++) {
			for (int j = 0; j <= len_of_str2; j++) {
				if (str1[i] == str2[i]) {
					return true;
				}
			}

		}

	}
	else {
		return false;
	}
	
}
