// 계산기 프로그램을 종료합니다.

#include <stdio.h>
#pragma warning (disable: 4996)
int main(void) {
	int menu; // 메뉴값을 저장하기 위한 변수
	int num1; // 첫번째 숫자를 저장하기 위한 변수
	int num2; // 두번째 숫자를 저장하기 위한 변수

	// 기능을 반복적으로 제공하기 위해 반복문 사용
	// 사용자가 종료를 언제 할지 모르므로 무한반복을 하면서 사용자에게 5번으로 종료를 유도한다.
	for (int i = 0; ; i++) {
		printf("기능을 선택해주세요. (1. 더하기, 2. 빼기, 3. 곱하기, 4. 나누기, 5. 종료 ) : ");
		scanf("%d", &menu);

		// 종료 후에 값을 입력받으면 안되기 때문에 숫자를 입력받는 곳 위에서 break
		if (menu == 5) {
			printf("계산기 프로그램을 종료합니다.");
			break;
		}
		// 숫자 2개를 입력받는 행위는 한번이면 충분하다.
		printf("첫번째 숫자를 입력해주세요 :");
		scanf("%d", &num1);
		printf("두번째 숫자를 입력해주세요 :");
		scanf("%d", &num2);

		if (menu == 1) {
			printf("결과: %d + %d = %d\n", num1, num2, num1 + num2);
		}
		else if (menu == 2) {
			printf("결과: %d - %d = %d\n", num1, num2, num1 - num2);
		}
		else if (menu == 3) {
			printf("결과: %d * %d = %d\n", num1, num2, num1 * num2);
		}
		else if (menu == 4) {
			printf("결과: %d / %d = %d\n", num1, num2, num1 / num2);
		}
		else {
			printf("그런 기능은 존재하지 않습니다.\n");
		}
	}

	return 0;
}

// scanf를 이용해 단어 하나를 word에 저장해주세요.
// 그리고 word가 love이면 I love you. 를 출력하세요. 아니면 출력 X

#include <stdio.h>
#pragma warning (disable: 4996)
#define true 1
#define false 0


bool get_str_len(char* str) {
    for (int i = 0; true; i++) {
        if (str[i] == '\0') {
            return i;
        }
    }
}

bool str_equals(char* str1, char* str2) {

    int len_of_str1 = get_str_len(str1);
    int len_of_str2 = get_str_len(str2);

    if (len_of_str1 != len_of_str2) {
        return false;
    }
    for (int i = 0; i < len_of_str1; i++) {
        if (str1[i] != str2[i]) {
            return false;
        }
    }
    return true;
}
