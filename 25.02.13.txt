// 문제 : 문장의 길이를 반환하는 함수를 만들어주세요.(get_str_len)
int get_str_len(char* name) {
	int sum = 0;
	for (int i = 0; 1; i++) { //여기서 1은 무한반복
		if (*(name + i) == '\0') {
			break;
		}
		sum++;
	} return sum;
}

int main(void) {
	char name[100] = "Paul"; // 이렇게 하면 name이 가리키는 배열에 Paul이 저장되면서 시작된다.
	int len = get_str_len(name);

	printf("len : %d\n", len);
	// 출력 => len : 4

	name[0] = 't';
	name[1] = 'o';
	name[2] = 'm';
	name[3] = 'a';
	name[4] = 's';
	name[5] = '\0';

	len = get_str_len(name);

	printf("len : %d\n", len);
	// 출력 => len : 5

	return 0;
}

// 문제 : 문장에서 특정 문자의 위치를 반환하는 함수를 만들어주세요.(get_index_of_c)

#include <stdio.h>

int get_index_of_c(char* str, char c) {
  int pos = -1;

  for (int i = 0; str[i] != '\0'; i++) {
    if (str[i] == c) {
      pos = i;
      break;
    }
  }

  return pos;
}

int main(void) {
  int index;

  index = get_index_of_c("abc", 'b');
  printf("index : %d\n", index);
  // 출력 => index : 1

  index = get_index_of_c("test", 's');
  printf("index : %d\n", index);
  // 출력 => index : 2

  index = get_index_of_c("test", 'x');
  printf("index : %d\n", index);
  // 출력 => index : -1

  return 0;
}

#include <stdio.h>

typedef int bool;

#define true 1

	for (int i = 1; str1[i] != '\0'; i++) {
		for (int i = 0; str2[i] != '\0'; i++) {
			if (str1[i] == str2[i]) {
				return 0;
			}
			else {
				return 1;
			}
		}
	}
}

	bool rs;

	rs = ends_with("abc", "bc");
	printf("rs : %d\n", rs); // 출력 rs : 1

	rs = ends_with("kbs", "kb");
